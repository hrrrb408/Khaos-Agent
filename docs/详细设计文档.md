```python
from dataclasses import dataclass, field
from enum import Enum
from typing import Optional
import asyncio


class StopReason(Enum):
    END_TURN = "end_turn"
    TOOL_USE = "tool_use"
    MAX_TURNS = "max_turns"
    MAX_BUDGET = "max_budget"
    USER_ABORT = "user_abort"
    ERROR = "error"


@dataclass
class AgentConfig:
    max_turns: int = 100          # 单轮对话最大工具调用次数
    max_budget_tokens: int = 500000  # 单轮对话最大 token 消耗
    stream_timeout: int = 120     # 流式响应超时（秒）
    compression_threshold: int = 128000  # 触发 auto-compact 的 token 阈值


@dataclass
class Message:
    role: str                    # "system" | "user" | "assistant" | "tool"
    content: str
    tool_calls: list[dict] = field(default_factory=list)
    tool_call_id: Optional[str] = None
    token_count: int = 0
    created_at: float = 0.0


class AgentLoop:
    """Agent 核心循环 (while true)"""

    def __init__(
        self,
        config: AgentConfig,
        mode_manager: "ModeManager",
        tool_registry: "ToolRegistry",
        permission_engine: "PermissionEngine",
        memory_manager: "MemoryManager",
        context_compressor: "ContextCompressor",
        router: "ModelRouter",
        token_engine,            # Rust FFI
        tool_executor,            # Rust FFI
        db: "Database",
    ):
        self.config = config
        self.mode_manager = mode_manager
        self.tool_registry = tool_registry
        self.permission_engine = permission_engine
        self.memory_manager = memory_manager
        self.compressor = context_compressor
        self.router = router
        self.token_engine = token_engine
        self.tool_executor = tool_executor
        self.db = db

    async def run(self, user_input: str, session_id: str) -> AsyncIterator[Message]:
        """
        主循环入口。返回流式 Message 迭代器。

        状态机: IDLE → BUILDING_CONTEXT → CALLING_MODEL → EXECUTING_TOOLS → COMPRESSING → DONE/ERROR

        消息落库时机: 每条消息（用户/助手/工具）立即写入 messages 表

        SSE 事件生成规则:
          - 模型返回 text chunk → SSE text 事件
          - 模型返回 tool_use → SSE tool_call 事件 → 检查权限 → SSE permission_request（如果需要确认）→ 执行工具 → SSE tool_result 事件
          - 循环直到 stop_reason != tool_use

        工具调用失败处理: failure 不算致命错误，ToolResult(success=False, error=...) 返回模型重决策，不抛异常

        上下文压缩检查点: 每次循环迭代前检查 total_tokens > threshold
        超预算退出: total_tokens > max_budget_tokens → 生成 done 事件
        """
        # === 状态机初始化 ===
        state = "IDLE"
        turn_count = 0
        total_tokens = 0

        try:
            # === IDLE → BUILDING_CONTEXT ===
            state = "BUILDING_CONTEXT"
            messages = await self._build_context(session_id)

            # 用户消息落库
            user_msg = Message(role="user", content=user_input)
            messages.append(user_msg)
            await self.db.insert_message(session_id, user_msg)  # 立即落库

            # === 主循环 ===
            while turn_count < self.config.max_turns:
                # 上下文压缩检查点：每次迭代前检查
                if await self._check_compression(messages):
                    state = "COMPRESSING"
                    result = await self.compressor.compress(
                        messages, self.config.compression_threshold
                    )
                    messages = result.messages
                    total_tokens = sum(m.token_count for m in messages)
                    # 压缩摘要消息落库
                    for msg in messages:
                        if msg.token_count > 0:
                            await self.db.update_message_tokens(msg)
                    state = "CALLING_MODEL"

                # 超预算退出
                if total_tokens > self.config.max_budget_tokens:
                    yield Message(
                        role="assistant",
                        content="",
                        token_count=total_tokens,
                    )
                    yield Message(
                        role="system",
                        content=f"demand: max_budget exceeded ({total_tokens} > {self.config.max_budget_tokens})",
                    )
                    return

                # === CALLING_MODEL ===
                state = "CALLING_MODEL"
                assistant_content = ""
                tool_calls = []
                stop_reason = None

                async for chunk in self.router.call("agent_loop", messages):
                    # text chunk → SSE text 事件
                    if chunk.content:
                        assistant_content += chunk.content
                        chunk.token_count = self.token_engine.count_tokens(chunk.content)
                        total_tokens += chunk.token_count
                        yield chunk  # SSE text 事件

                    if chunk.tool_calls:
                        tool_calls.extend(chunk.tool_calls)

                    if hasattr(chunk, "stop_reason") and chunk.stop_reason:
                        stop_reason = chunk.stop_reason

                turn_count += 1

                # 助手消息落库（包含 tool_calls）
                assistant_msg = Message(
                    role="assistant",
                    content=assistant_content,
                    tool_calls=tool_calls,
                    token_count=self.token_engine.count_tokens(assistant_content),
                )
                messages.append(assistant_msg)
                await self.db.insert_message(session_id, assistant_msg)  # 立即落库

                # === 判断 stop_reason ===
                if stop_reason != "tool_use":
                    # end_turn / max_turns / error → 退出
                    state = "DONE"
                    return

                # === EXECUTING_TOOLS ===
                state = "EXECUTING_TOOLS"
                for tool_result_msg in await self._execute_tool_calls(tool_calls):
                    # tool_result → SSE tool_result 事件
                    yield tool_result_msg
                    messages.append(tool_result_msg)
                    total_tokens += tool_result_msg.token_count
                    await self.db.insert_message(session_id, tool_result_msg)  # 立即落库

            # max_turns 退出
            state = "DONE"

        except asyncio.CancelledError:
            state = "ERROR"
            raise

        except Exception as e:
            state = "ERROR"
            logger.error("Agent loop error: %s", e, exc_info=True)
            yield Message(role="system", content=f"error: {e}")


    async def _build_context(self, session_id: str) -> list[Message]:
        """组装上下文: system prompt + 记忆注入 + 上下文文件 + 历史消息"""
        ...

    async def _call_model(
        self, messages: list[Message]
    ) -> AsyncIterator[Message]:
        """调用模型 API，流式返回 Message chunks"""
        ...

    async def _execute_tool_calls(
        self, tool_calls: list[dict]
    ) -> list[Message]:
        """
        批量执行工具调用。
        只读工具并行 (Rust executor)，写操作串行。
        每个工具调用前经过权限检查。
        """
        ...

    async def _check_compression(self, messages: list[Message]) -> bool:
        """检查是否需要触发上下文压缩"""
        total_tokens = sum(m.token_count for m in messages)
        return total_tokens > self.config.compression_threshold
```

#### 1.1.3 compressor.py — 上下文压缩

```python
from enum import Enum
from dataclasses import dataclass


class CompressionLevel(Enum):
    MICRO_COMPACT = 0      # 单条消息截断
    CONTEXT_COLLAPSE = 1   # 折叠为摘要
    AUTO_COMPACT = 2       # 模型压缩
    SESSION_MEMORY = 3     # 持久化到记忆


@dataclass
class CompressionResult:
    level: CompressionLevel
    original_tokens: int
    compressed_tokens: int
    messages: list[Message]  # 压缩后的消息列表


class ContextCompressor:
    """三层上下文压缩 + 熔断机制"""

    def __init__(self, router: "ModelRouter", token_engine, memory_manager: "MemoryManager"):
        self.router = router
        self.token_engine = token_engine
        self.memory_manager = memory_manager
        self._consecutive_failures = 0

    async def compress(
        self, messages: list[Message], threshold: int
    ) -> CompressionResult:
        """
        按阈值自动选择压缩层级:
        - token > threshold * 1.5 → Level 2 (模型压缩)
        - token > threshold * 1.2 → Level 1 (摘要折叠)
        - 单条消息 > 10000 chars → Level 0 (截断)

        压缩边界保护：
          - tool_call 和对应的 tool_result 必须成对保留，不能只压缩其中一个
          - 最近 4 条消息（2轮对话）始终保留完整
          - system prompt 不参与压缩

        摘要格式：[摘要开始] 原始 N 条消息压缩为 1 条。包含关键决策: ... [摘要结束]

        失败回退：
          - Level 2 模型压缩失败 → 降级到 Level 1 context-collapse
          - Level 1 失败 → 降级到 Level 0 micro-compact（粗暴截断最早的消息）
          - 连续 3 次 Level 2 失败 → 熔断器打开，后续只用 Level 0/1
        """
        original_tokens = sum(self.token_engine.count_tokens(m.content) for m in messages)

        # === 压缩边界保护 ===
        # 分离不可压缩的消息
        system_msgs = [m for m in messages if m.role == "system"]
        recent_msgs = messages[-4:] if len(messages) > 4 else []  # 最近 4 条（2轮）
        compressible_start = len(system_msgs)
        compressible_end = len(messages) - len(recent_msgs)
        compressible = messages[compressible_start:compressible_end]

        if len(compressible) <= 1:
            return CompressionResult(
                level=CompressionLevel.MICRO_COMPACT,
                original_tokens=original_tokens,
                compressed_tokens=original_tokens,
                messages=messages,
            )

        # === Level 0: 单条消息截断（零 API 开销） ===
        for msg in compressible:
            if len(msg.content) > 10000:
                msg = await self._micro_compact(msg)
                messages[compressible_start] = msg

        total_tokens = sum(self.token_engine.count_tokens(m.content) for m in messages)
        if total_tokens <= threshold:
            return CompressionResult(
                level=CompressionLevel.MICRO_COMPACT,
                original_tokens=original_tokens,
                compressed_tokens=total_tokens,
                messages=messages,
            )

        # === 确定压缩层级 ===
        if total_tokens > threshold * 1.5 and not self.is_circuit_open:
            # Level 2: 模型压缩
            try:
                collapsed = await self._auto_compact(compressible)
                # 合并：system + collapsed + recent
                messages = system_msgs + collapsed + recent_msgs
                self._consecutive_failures = 0  # 成功，重置计数
                compressed_tokens = sum(
                    self.token_engine.count_tokens(m.content) for m in messages
                )
                return CompressionResult(
                    level=CompressionLevel.AUTO_COMPACT,
                    original_tokens=original_tokens,
                    compressed_tokens=compressed_tokens,
                    messages=messages,
                )
            except Exception as e:
                # Level 2 失败 → 降级到 Level 1
                logger.warning("Level 2 compression failed: %s, falling back to Level 1", e)
                self._consecutive_failures += 1
                if self._consecutive_failures >= 3:
                    logger.error("Circuit breaker OPEN: 3 consecutive Level 2 failures")

        # === Level 1: 折叠为摘要 ===
        try:
            collapsed = await self._context_collapse(compressible)
            messages = system_msgs + collapsed + recent_msgs
            compressed_tokens = sum(
                self.token_engine.count_tokens(m.content) for m in messages
            )
            return CompressionResult(
                level=CompressionLevel.CONTEXT_COLLAPSE,
                original_tokens=original_tokens,
                compressed_tokens=compressed_tokens,
                messages=messages,
            )
        except Exception as e:
            # Level 1 失败 → 降级到 Level 0（粗暴截断）
            logger.warning("Level 1 compression failed: %s, falling back to Level 0", e)
            # Level 0: 截断最早的 compressible 消息
            truncated = compressible[:1]
            for msg in truncated:
                msg = await self._micro_compact(msg, max_chars=2000)
            messages = system_msgs + [truncated[0]] + recent_msgs
            compressed_tokens = sum(
                self.token_engine.count_tokens(m.content) for m in messages
            )
            return CompressionResult(
                level=CompressionLevel.MICRO_COMPACT,
                original_tokens=original_tokens,
                compressed_tokens=compressed_tokens,
                messages=messages,
            )

    async def _micro_compact(self, message: Message, max_chars: int = 10000) -> Message:
        """
        Level 0: 截断单条过长消息，零 API 开销。
        在 max_chars 处截断，附加截断标记。
        """
        if len(message.content) <= max_chars:
            return message
        truncated = message.content[:max_chars]
        return Message(
            role=message.role,
            content=f"{truncated}\n\n[截断: 原文 {len(message.content)} 字符，已保留前 {max_chars}]",
            tool_calls=message.tool_calls,
            tool_call_id=message.tool_call_id,
        )

    async def _context_collapse(self, messages: list[Message]) -> list[Message]:
        """
        Level 1: 折叠历史消息为摘要（保留最近 N 条完整）。
        将可压缩消息折叠为一条摘要消息。
        确保工具调用/工具结果成对保留。
        """
        # 收集成对的 tool_call + tool_result，确保不拆散
        protected_pairs = []
        unprotected = []
        i = 0
        while i < len(messages):
            msg = messages[i]
            if msg.tool_calls:
                # 查找对应的 tool result
                tool_call_ids = [tc.get("id") for tc in msg.tool_calls]
                tool_results = []
                j = i + 1
                while j < len(messages) and len(tool_results) < len(tool_call_ids):
                    if messages[j].role == "tool" and messages[j].tool_call_id in tool_call_ids:
                        tool_results.append(messages[j])
                        j += 1
                    else:
                        break
                # 成对保留
                protected_pairs.append(msg)
                protected_pairs.extend(tool_results)
                i = j
            else:
                unprotected.append(msg)
                i += 1

        # 将 unprotected 消息折叠为摘要
        if not unprotected:
            return messages

        summary_content = (
            f"[摘要开始] 原始 {len(unprotected)} 条消息压缩为 1 条。"
            f"包含关键决策: {extract_key_decisions(unprotected)} [摘要结束]"
        )
        summary_msg = Message(role="assistant", content=summary_content)

        return protected_pairs + [summary_msg]

    async def _auto_compact(self, messages: list[Message]) -> list[Message]:
        """
        Level 2: 调用模型压缩对话。连续 3 次失败触发熔断。
        将消息列表发给压缩模型，生成摘要。
        确保工具调用/工具结果成对保留。
        """
        # 构建压缩请求
        compress_prompt = (
            "请将以下对话历史压缩为简洁的摘要，保留关键决策、"
            "工具调用及其结果、重要的事实和数字。"
            "格式要求：[摘要开始] 原始 N 条消息压缩为 1 条。"
            "包含关键决策: ... [摘要结束]\n\n"
            + "\n".join(
                f"[{m.role}]: {m.content[:500]}{'...' if len(m.content) > 500 else ''}"
                for m in messages
            )
        )

        # 调用压缩模型
        response = await self.router.call("compression", [Message(role="user", content=compress_prompt)])
        summary = "".join(chunk.content async for chunk in response)

        return [Message(role="assistant", content=summary)]

    @property
    def is_circuit_open(self) -> bool:
        """熔断器是否打开"""
        return self._consecutive_failures >= 3

    def _reset_circuit(self):
        self._consecutive_failures = 0
```

#### 1.3.2 scheduler.py — 工具调度

```python
from dataclasses import dataclass
from typing import Any


@dataclass
class ToolResult:
    tool_call_id: str
    name: str
    success: bool
    output: Any
    error: str = ""
    duration_ms: int = 0
    arguments: dict[str, Any] | None = None  # 原始调用参数，供任务追踪使用


@dataclass
class ToolBudget:
    max_calls: int = 50              # 最大工具调用次数
    max_output_chars: int = 100000   # 最大工具输出总字符数
    _call_count: int = 0
    _output_chars: int = 0

    @property
    def is_exhausted(self) -> bool:
        return self._call_count >= self.max_calls or self._output_chars >= self.max_output_chars

    def record(self, output_chars: int):
        self._call_count += 1
        self._output_chars += output_chars


class ToolScheduler:
    """工具调度器 — 并行/串行分发"""

    def __init__(
        self,
        registry: ToolRegistry,
        permission_engine: "PermissionEngine",
        tool_executor,          # Rust FFI
        budget: ToolBudget,
    ):
        self.registry = registry
        self.permission_engine = permission_engine
        self.tool_executor = tool_executor
        self.budget = budget

    async def execute_batch(
        self, tool_calls: list[dict], mode: str
    ) -> list[ToolResult]:
        """
        批量执行工具调用。

        具体流程：
          1. 预算检查：budget.is_exhausted → 返回空结果 + 警告
          2. 遍历 tool_calls，逐个调用 permission_engine.check()
             - 返回 "deny" → 跳过，返回 ToolResult(success=False, error="denied")
             - 返回 "ask" → 暂停批量，发送 permission_request SSE 事件，等用户确认后继续
          3. 权限全部通过后，调用 registry.get_parallel_tools() 分组
          4. 并行组 → rust_tool_executor.parallel(calls, timeout)
          5. 串行组 → 依次调用，每个结果检查 success，记录预算
          6. 如果 budget 耗尽 → 中断剩余工具调用，返回已完成的 + 警告

        失败策略：单个工具失败不中断其他工具
        """
        results: list[ToolResult] = []

        # === 1. 预算检查 ===
        if self.budget.is_exhausted:
            logger.warning("Tool budget exhausted, skipping all %d tool calls", len(tool_calls))
            return results

        # === 2. 权限检查（逐个） ===
        approved_calls: list[dict] = []
        for call in tool_calls:
            tool_name = call["name"]
            params = call.get("arguments", {})
            tool_def = self.registry.get(tool_name)

            decision = await self.permission_engine.check(
                tool_name=tool_name,
                params=params,
                permission_level=tool_def.permission_level,
                mode=mode,
            )

            if decision.approved.value == "deny":
                # 拒绝 → 跳过，返回 denied 结果
                results.append(ToolResult(
                    tool_call_id=call["id"],
                    name=tool_name,
                    success=False,
                    error=f"Permission denied: {decision.reason}",
                ))
            elif decision.requires_user_confirm:
                # ask → 暂停批量，发送 permission_request SSE 事件
                # 等 HTTP POST /api/chat/:id/confirm 后继续
                yield_sse_event("permission_request", {
                    "id": call["id"],
                    "name": tool_name,
                    "arguments": params,
                    "level": tool_def.permission_level,
                })
                confirmed = await wait_for_user_confirm(call["id"])
                if not confirmed:
                    results.append(ToolResult(
                        tool_call_id=call["id"],
                        name=tool_name,
                        success=False,
                        error="User denied permission",
                    ))
                    continue
                # 用户确认，如果带 remember=True → 生成并持久化规则
                if confirmed.get("remember"):
                    await self.permission_engine.grant_rule(
                        PermissionRule(
                            pattern=normalize_target(tool_name, params),
                            permission_level=tool_def.permission_level,
                            approval=decision.approved,
                            mode=mode,
                        )
                    )
                approved_calls.append(call)
            else:
                # allow → 直接加入
                approved_calls.append(call)

        if not approved_calls:
            return results

        # === 3. 分组: 并行组 + 串行组 ===
        parallel_calls, serial_calls = self.registry.get_parallel_tools(approved_calls)

        # === 4. 并行组 → Rust executor 并发执行 ===
        if parallel_calls:
            try:
                parallel_results = self.tool_executor.execute_tools_parallel(
                    parallel_calls,
                    timeout_ms=get_global_timeout(),
                )
                for i, pr in enumerate(parallel_results):
                    tool_result = ToolResult(
                        tool_call_id=parallel_calls[i]["id"],
                        name=parallel_calls[i]["name"],
                        success=pr.success,
                        output=pr.output,
                        error=pr.error or "",
                    )
                    results.append(tool_result)
                    self.budget.record(len(str(pr.output)))
            except Exception as e:
                # 整组失败 → 为每个调用生成失败结果
                for call in parallel_calls:
                    results.append(ToolResult(
                        tool_call_id=call["id"],
                        name=call["name"],
                        success=False,
                        error=f"Parallel execution failed: {e}",
                    ))

        # === 5. 串行组 → 依次执行 ===
        for call in serial_calls:
            if self.budget.is_exhausted:
                # 预算耗尽 → 中断剩余
                logger.warning("Budget exhausted during serial execution, skipping %s", call["name"])
                results.append(ToolResult(
                    tool_call_id=call["id"],
                    name=call["name"],
                    success=False,
                    error="Tool budget exhausted",
                ))
                break

            try:
                tool_def = self.registry.get(call["name"])
                output = await tool_def.handler(**call.get("arguments", {}))
                result = ToolResult(
                    tool_call_id=call["id"],
                    name=call["name"],
                    success=True,
                    output=output,
                )
                self.budget.record(len(str(output)))
                results.append(result)
            except Exception as e:
                # 单个工具失败 → 不中断其他工具
                results.append(ToolResult(
                    tool_call_id=call["id"],
                    name=call["name"],
                    success=False,
                    error=str(e),
                ))

        return results

    async def _check_permission(
        self, tool_name: str, params: dict, mode: str
    ) -> str:
        """检查权限，返回 "allow" | "deny" | "ask" """
        ...
```

#### 1.5.1 engine.py

```python
from enum import Enum
from dataclasses import dataclass


class ApprovalMode(Enum):
    AUTO_APPROVE = "auto-approve"
    SUGGEST = "suggest"
    ASK_EVERY = "ask-every"
    DENY = "deny"


@dataclass
class PermissionRule:
    id: Optional[int]
    pattern: str              # glob 模式 (如 "npm install *")
    permission_level: str     # "read" | "write" | "execute" | "network"
    approval: ApprovalMode     # 该模式下的审批结果
    mode: str                 # "office" | "coding" | "all"
    granted_at: float = 0.0


@dataclass
class PermissionDecision:
    approved: ApprovalMode
    reason: str                # 审批理由（展示给用户）
    matched_rule: Optional[PermissionRule] = None
    requires_user_confirm: bool = False


class PermissionEngine:
    """权限引擎 — 规则匹配 + 用户确认"""

    def __init__(self, db: "Database", default_mode: ApprovalMode = ApprovalMode.ASK_EVERY):
        self.db = db
        self._default_mode = default_mode
        self._rules: list[PermissionRule] = []

    async def load_rules(self) -> None:
        """从数据库加载权限规则"""
        ...

    async def check(
        self,
        tool_name: str,
        params: dict,
        permission_level: str,
        mode: str,
    ) -> PermissionDecision:
        """
        权限检查。

        target 规范化规则：
          - 文件路径：os.path.normpath() + os.path.realpath() 解析后匹配
          - 命令：拆解为 base_command + args，base_command 匹配
          - 网络 URL：解析为 scheme + host 匹配

        glob 匹配：fnmatch.fnmatch()，支持 * ? [seq] [!seq]
        规则遍历顺序：按 granted_at DESC（最新规则优先）
        默认返回：无匹配规则 → PermissionDecision(approved=ApprovalMode.ASK_EVERY, requires_user_confirm=True)

        remember rule 生成：用户确认时如果带 remember=True → 生成 PermissionRule(pattern=规范化后的target, ...) 并持久化
        """
        # === target 规范化 ===
        target = self._normalize_target(tool_name, params)

        # === 遍历规则表（按 granted_at DESC，最新规则优先） ===
        for rule in self._rules:
            # 模式隔离检查：如果规则指定了模式，必须匹配当前模式
            if rule.mode != "all" and rule.mode != mode:
                continue

            # 权限等级检查：规则等级必须匹配或包含
            if rule.permission_level != permission_level:
                continue

            # glob 模式匹配
            if self._match_pattern(rule.pattern, target):
                # 匹配到规则 → 返回对应审批结果
                return PermissionDecision(
                    approved=rule.approval,
                    reason=f"Matched rule: {rule.pattern} (granted at {rule.granted_at})",
                    matched_rule=rule,
                    requires_user_confirm=(rule.approval == ApprovalMode.ASK_EVERY),
                )

        # === 无匹配规则 → 使用默认模式 ===
        if self._default_mode == ApprovalMode.AUTO_APPROVE:
            return PermissionDecision(
                approved=ApprovalMode.AUTO_APPROVE,
                reason="No matching rule, default: auto-approve",
                requires_user_confirm=False,
            )
        elif self._default_mode == ApprovalMode.DENY:
            return PermissionDecision(
                approved=ApprovalMode.DENY,
                reason="No matching rule, default: deny",
                requires_user_confirm=False,
            )
        else:
            # 默认: ask-every
            return PermissionDecision(
                approved=ApprovalMode.ASK_EVERY,
                reason="No matching rule, default: ask-every",
                requires_user_confirm=True,
            )

    async def grant_rule(self, rule: PermissionRule) -> None:
        """用户授权后，持久化权限规则"""
        ...

    async def revoke_rule(self, rule_id: int) -> None:
        """撤销权限规则"""
        ...

    def _match_pattern(self, pattern: str, target: str) -> bool:
        """glob 模式匹配。使用 fnmatch.fnmatch()，支持 * ? [seq] [!seq]"""
        return fnmatch.fnmatch(target, pattern)
```

## 5.5 运行时后台服务与平台 Webhook 安全

- `AgentService` 持有进程级 `CronEngine`；JSON-line 服务监听前调用
  `AgentService.start()`，服务退出时在 `finally` 中调用 `shutdown()`。
- Cron 任务通过普通 office-mode Agent 主链路执行，使用 `cron:<task_id>` 作为
  会话 ID；引擎实例注入 `cron_tools`，确保 RPC、Agent 工具共享同一任务状态。
- Discord 入站请求使用 Ed25519 验证 `timestamp + raw_body`，配置的 secret
  必须为 Discord application public key 的十六进制形式。
- 微信 XML 使用 `xml.etree.ElementTree` 解析，并兼容命名空间标签；禁止使用
  正则表达式解释 XML 结构。

## 5.6 剩余风险收敛设计

- 代码理解：Python 使用 AST；Go/Rust 使用零依赖结构化正则提取函数、调用、
  `import`、`mod` 和 `use` 关系。解析失败必须降级为空图，不影响 Agent 主链路。
- 长任务：`coding_tasks` 保存完整 JSON 状态。服务启动恢复任务，并将重启前的
  活跃状态转成 `blocked`，错误注明 `interrupted by process restart`。
- 执行隔离：macOS 终端通过 `sandbox-exec`，Linux 通过 `bwrap`；若父进程已处于
  Codex OS 沙箱则继承父级约束。没有可用 OS 沙箱时拒绝执行。交互式 Python、
  `python -c` 以及非测试用途的 `python -m` 均直接阻断。
- 自动化串联：AgentLoop 持久化消息后同步写入 `messages_fts`；工具结果写入任务
  trace；成功任务结束时调用 `SkillGenerator` 并把候选存入任务 metadata。
- Bot 出站：四个平台统一使用注入式 `httpx.AsyncClient`。生产环境执行真实 HTTP，
  测试必须使用 `MockTransport`，禁止访问真实平台。

## 5.7 GPT 审查问题收敛

- CLI、TUI、Gateway 和 SubAgent 必须通过异步 `runtime.build_runtime()` 构造
  `AgentLoop`；Factory 负责加载模式、权限、记忆、技能、安全策略及 Coding 组件。
- 每次 Coding turn 使用独立 `VerifyFixLoop`，任务结束依据 stop reason 写入
  completed、failed 或 cancelled；终止态不得退回活跃态。
- 工具输出在进入 `ToolResult` 前完成 secret 扫描和递归脱敏，私钥块整体替换。
- GitHub 远程操作独立于本地 Git 工具，通过异步 `gh` 子进程实现 PR、Issue、
  评论和 reviewer 工作流。
- Go 网关通过 `/v1/tasks` 暴露创建、查询、审批、拒绝、取消、artifact 与 SSE

## 5.8 GPT 二次审查收敛

## 6. 错误码定义

## 5.9 Coding Runtime 对齐实现边界

Coding runtime 的新增能力采用独立适配层：Worktree/ChangeSet 负责隔离与审批绑定，ExecutionBackend 负责统一执行边界，LanguageRegistry/IndexStore 负责结构化代码检索，VerificationPipeline 负责分层验证。Legacy Parser、`test_run`、Verify-Fix 和既有权限/沙箱入口继续保留。

### 5.9.2 Implementation Planning（M4 Batch 0）

`coding/planning` 提供不可变的 `ImplementationPlan`、影响/验证/风险/诊断/证据模型以及只读 `PlanningService` 契约。计划只能基于 `CodeQueryService`、Resolution graph、`IndexStore` 和受信任仓库元数据生成；每个识别出的文件或 Symbol 都绑定可复核证据。服务不持有 `ToolScheduler`、ExecutionBackend、Worktree、ChangeSet 或 ApprovalBroker，因此不能写入用户仓库、执行命令、创建 ChangeSet 或自行审批。

相同仓库元数据、查询结果、目标和配置会产生相同的规范化目标、影响范围、验证要求、风险和 content hash；时间戳不参与 hash。HEAD、repository generation、文件 hash 或证据 Symbol 漂移会使计划变为 stale，stale 计划不得进入执行阶段。动态/歧义/缺失的解析结果必须降级为 diagnostic，且不得编造 stable symbol ID。验证命令只能来自受信任仓库元数据或既有验证管线，不能由目标文本或未来 LLM 候选直接指定。

### 5.9.3 Planned Workspace Mutation（M4 Batch 3.0）

计划执行只允许通过 `ApprovalRuntime → PlannedExecutionGuard →
WorkspaceMutationEngine`。输入是服务端重新计算摘要的结构化 Edit Bundle，操作限制为
create、update、delete、rename；执行根固定为 active `TaskWorkspace.worktree_path`，
不得接受调用方提供的任意 filesystem root。每次 edit 都重新验证 active Context、Lease、
Boot、Task/Workspace、Plan binding 和 mutation fence。

文件修改采用同目录临时文件、flush/fsync、受控 mode、原子 replace 和父目录 fsync。
修改前必须先把路径、hash、mode、状态及恢复 artifact 标识写入 SQLite Journal；源码正文
只允许进入 Worktree 外的私有 `.khaos-recovery/<execution_run_id>/`，不得进入数据库、
Audit 或普通日志。失败时按 Journal 逆序 rollback；无法证明恢复完整时 poison Workspace，
后续 acquire/index/cleanup/planned mutation 全部 fail closed，直到受控 recovery 完成。

本阶段不执行 verification、Tool、terminal/test_run，不生成或应用正式 ChangeSet，也不做
Git commit/push。现有 M2 file tools、WorkspaceManager ChangeSet 和 Git 工具继续遵循各自
权限、沙箱与审批规则，不能被 planned execution capability 替代。

### 5.9.1 LanguageRegistry 与解析契约

`coding/intelligence` 内的解析结果统一使用 `ParseResult`。内部行号和列号均为
0-based；列号表示 Unicode code point 数，不是 UTF-8 byte column，也不是 LSP
使用的 UTF-16 code unit。`byte_start` 和 `byte_end` 是原始内容 UTF-8 bytes 的
半开区间。旧接口的 `line` 字段继续通过兼容层输出 1-based 值。

语言只由锁定的扩展名注册表解析，不接受用户提供的任意语言字符串。Tree-sitter
及 grammar 都是可选的锁定依赖或预构建包：运行时不得联网安装、不得从用户项目
目录搜索 grammar、不得动态加载项目内 native library。Python 的降级顺序为
Tree-sitter、stdlib AST、Legacy Regex；JavaScript、TypeScript、Go、Rust 为
Tree-sitter、Legacy Regex。任何可恢复失败都写入 structured diagnostic，不能
中止仓库级索引。

可选依赖不可用时必须返回降级诊断，不得运行时联网安装；可写 Task 默认只使用独立 Worktree，主工作树回写必须经过 ChangeSet 审批。
